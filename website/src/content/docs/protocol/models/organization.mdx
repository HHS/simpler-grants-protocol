An organization that can apply to or host funding opportunities.

### Table

| Property    | Type                                                                        | Required | Description                                                            |
| ----------- | --------------------------------------------------------------------------- | -------- | ---------------------------------------------------------------------- |
| name        | [string](/protocol/types/string#string)                                     | Yes      | The organization's legal name as registered with relevant authorities. |
| address     | [Address](/protocol/fields/address)                                         | Yes      | The organization's primary mailing address.                            |
| type        | [string](/protocol/types/string#string)                                     | Yes      | The organization's type (e.g., "Nonprofit", "For-Profit", etc.)        |
| ein         | [string](/protocol/types/string#string)                                     | No       | The organization's Employer Identification Number (EIN).               |
| uei         | [string](/protocol/types/string#string)                                     | No       | The organization's Unique Entity Identifier (UEI).                     |
| duns        | [string](/protocol/types/string#string)                                     | No       | The organization's Data Universal Numbering System (DUNS) number.      |
| mission     | [string](/protocol/types/string#string)                                     | No       | The organization's mission statement.                                  |
| yearFounded | [integer](/protocol/types/numeric#integer)                                  | No       | The calendar year the organization was founded.                        |
| socials     | [Record](/protocol/types/other#record)\<[OrgSocialLinks](#orgsociallinks)\> | No       | The organization's social media and web presence links.                |

model Organization {

    duns?: string;

    /** The organization's mission statement. */
    mission?: string;

    /** The calendar year the organization was founded. */
    yearFounded?: integer;

    /** The start date of the organization's fiscal year, used for financial reporting. */
    @example(plainDate.fromISO("2024-01-01"))
    fiscalYearStart?: plainDate;

    fiscalYearEnd?: plainDate;
